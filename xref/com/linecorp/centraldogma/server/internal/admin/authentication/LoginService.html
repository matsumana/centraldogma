59  /**
60   * A service to handle a login request to Central Dogma Web admin service.
61   */
62  public class LoginService extends AbstractHttpService {
63  
64      private static final Logger logger = LoggerFactory.getLogger(LoginService.class);
65  
66      private final CentralDogmaSecurityManager securityManager;
67      private final CommandExecutor executor;
68      private final Function<String, String> loginNameNormalizer;
69      private final Cache<String, AccessToken> cache;
70  
71      public LoginService(CentralDogmaSecurityManager securityManager, CommandExecutor executor,
72                          Function<String, String> loginNameNormalizer, Cache<String, AccessToken> cache) {
73          this.securityManager = requireNonNull(securityManager, "securityManager");
74          this.executor = requireNonNull(executor, "executor");
75          this.loginNameNormalizer = requireNonNull(loginNameNormalizer, "loginNameNormalizer");
76          this.cache = requireNonNull(cache, "cache");
77      }
78  
79      @Override
80      protected HttpResponse doPost(ServiceRequestContext ctx, HttpRequest req) throws Exception {
81          final CompletableFuture<HttpResponse> future = new CompletableFuture<>();
82          req.aggregate().thenAccept(aMsg -> {
83              final UsernamePasswordToken usernamePassword;
84              try {
85                  usernamePassword = usernamePassword(aMsg);
86              } catch (HttpResponseException e) {
87                  future.complete(e.httpResponse());
88                  return;
89              }
90  
91              ctx.blockingTaskExecutor().execute(() -> {
92                  // Need to set the thread-local security manager to silence
93                  // the UnavailableSecurityManagerException logged at DEBUG level.
94                  ThreadContext.bind(securityManager);
95                  Subject currentUser = null;
96                  boolean success = false;
97                  try {
98                      final AccessToken currentUserToken = currentUserTokenIfPresent(usernamePassword);
99                      if (currentUserToken != null) {
100                         future.complete(HttpResponse.of(HttpStatus.OK, MediaType.JSON_UTF_8,
101                                                         Jackson.writeValueAsBytes(currentUserToken)));
102                         return;
103                     }
104 
105                     currentUser = new Subject.Builder(securityManager).buildSubject();
106                     currentUser.login(usernamePassword);
107 
108                     final Session currentUserSession = currentUser.getSession(false);
109                     final long expiresIn = currentUserSession.getTimeout();
110                     final String sessionId = currentUserSession.getId().toString();
111                     final SimpleSession session = securityManager.getSerializableSession(sessionId);
112                     executor.execute(Command.createSession(session)).join();
113                     success = true;
114 
115                     logger.info("{} Logged in: {} ({})", ctx, usernamePassword.getUsername(), sessionId);
116 
117                     final AccessToken accessToken = new AccessToken(sessionId, expiresIn);
118                     cache.put(usernamePassword.getUsername(), accessToken);
119                     future.complete(HttpResponse.of(HttpStatus.OK, MediaType.JSON_UTF_8,
120                                                     Jackson.writeValueAsBytes(accessToken)));
121                 } catch (IncorrectCredentialsException e) {
122                     // Not authorized
123                     logger.debug("{} Incorrect login: {}", ctx, usernamePassword.getUsername());
124                     future.complete(HttpApiUtil.newResponse(HttpStatus.UNAUTHORIZED, "Incorrect login"));
125                 } catch (Throwable t) {
126                     logger.warn("{} Failed to authenticate: {}", ctx, usernamePassword.getUsername(), t);
127                     future.complete(HttpApiUtil.newResponse(HttpStatus.INTERNAL_SERVER_ERROR, t));
128                 } finally {
129                     try {
130                         if (!success && currentUser != null) {
131                             // Delete failed session.
132                             currentUser.logout();
133                         }
134                     } finally {
135                         ThreadContext.unbindSecurityManager();
136                     }
137                 }
138             });
139         }).exceptionally(voidFunction(cause -> {
140             logger.warn("{} Unexpected exception:", ctx, cause);
141             future.complete(HttpApiUtil.newResponse(HttpStatus.INTERNAL_SERVER_ERROR, cause));
142         }));
143         return HttpResponse.from(future);
144     }
145 
146     /**
147      * Returns {@link UsernamePasswordToken} which holds a username and a password.
148      */
149     private UsernamePasswordToken usernamePassword(AggregatedHttpMessage req) {
150         // check the Basic HTTP authentication first (https://tools.ietf.org/html/rfc7617)
151         final BasicToken basicToken = AuthTokenExtractors.BASIC.apply(req.headers());
152         if (basicToken != null) {
153             return new UsernamePasswordToken(basicToken.username(), basicToken.password());
154         }
155 
156         final MediaType mediaType = req.headers().contentType();
157         if (mediaType != MediaType.FORM_DATA) {
158             return throwResponse(HttpStatus.BAD_REQUEST,
159                                  "The content type of a login request must be '%s'.", MediaType.FORM_DATA);
160         }
161 
162         final Map<String, List<String>> parameters = new QueryStringDecoder(
163                 req.content().toStringUtf8(), false).parameters();
164 
165         // assume that the grant_type is "password"
166         final List<String> usernames = parameters.get("username");
167         final List<String> passwords = parameters.get("password");
168         if (usernames != null && passwords != null) {
169             final String username = usernames.get(0);
170             final String password = passwords.get(0);
171             return new UsernamePasswordToken(loginNameNormalizer.apply(username), password);
172         }
173 
174         return throwResponse(HttpStatus.BAD_REQUEST, "A login request must contain username and password.");
175     }
176 
177     @Nullable
178     private AccessToken currentUserTokenIfPresent(UsernamePasswordToken usernamePassword) {
179         securityManager.authenticate(usernamePassword);
180         // Because securityManager.authenticate does not throw any Exception, the user is authenticated.
181         final AccessToken currentUserToken = cache.getIfPresent(usernamePassword.getUsername());
182 
183         if (currentUserToken != null) {
184             final long currentTimeMillis = System.currentTimeMillis();
185             if (currentUserToken.deadline() >
186                 currentTimeMillis + Math.min(securityManager.globalSessionTimeout(), 60000 /* 1 minute */)) {
187                 return new AccessToken(currentUserToken.accessToken(),
188                                        currentUserToken.deadline() - currentTimeMillis);
189             }
190         }
191         return null;
192     }
