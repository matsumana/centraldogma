37  /**
38   * Utility methods that are useful when implementing a {@link Repository} implementation.
39   */
40  final class RepositoryUtil {
41  
42      static final Map<FindOption<?>, Object> EXISTS_FIND_OPTIONS =
43              ImmutableMap.of(FindOption.FETCH_CONTENT, false, FindOption.MAX_ENTRIES, 1);
44      static final Map<FindOption<?>, Object> GET_FIND_OPTIONS = ImmutableMap.of(FindOption.MAX_ENTRIES, 1);
45  
46      private static final CancellationException CANCELLATION_EXCEPTION =
47              Exceptions.clearTrace(new CancellationException("parent complete"));
48  
49      static <T> CompletableFuture<Entry<T>> watch(Repository repo, Revision lastKnownRev, Query<T> query) {
50  
51          requireNonNull(repo, "repo");
52          requireNonNull(lastKnownRev, "lastKnownRev");
53          requireNonNull(query, "query");
54  
55          final Query<Object> castQuery = unsafeCast(query);
56          final CompletableFuture<Entry<Object>> parentFuture = new CompletableFuture<>();
57          repo.getOrNull(lastKnownRev, castQuery)
58              .thenAccept(oldResult -> watch(repo, castQuery, lastKnownRev, oldResult, parentFuture))
59              .exceptionally(voidFunction(parentFuture::completeExceptionally));
60  
61          return unsafeCast(parentFuture);
62      }
63  
64      private static void watch(Repository repo, Query<Object> query,
65                                Revision lastKnownRev, @Nullable Entry<Object> oldResult,
66                                CompletableFuture<Entry<Object>> parentFuture) {
67  
68          final CompletableFuture<Revision> future = repo.watch(lastKnownRev, query.path());
69          parentFuture.whenComplete((res, cause) -> future.completeExceptionally(CANCELLATION_EXCEPTION));
70  
71          future.thenCompose(newRev -> repo.getOrNull(newRev, query).thenAccept(newResult -> {
72              if (newResult == null ||
73                  oldResult != null && Objects.equals(oldResult.content(), newResult.content())) {
74                  // Entry does not exist or did not change; watch again for more changes.
75                  if (!parentFuture.isDone()) {
76                      // ... only when the parent future has not been cancelled.
77                      watch(repo, query, newRev, oldResult, parentFuture);
78                  }
79              } else {
80                  parentFuture.complete(newResult);
81              }
82          })).exceptionally(voidFunction(parentFuture::completeExceptionally));
83      }
84  
85      private RepositoryUtil() {}
86  }
